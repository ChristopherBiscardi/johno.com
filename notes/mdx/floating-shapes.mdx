import * as THREE from 'three'
import { Canvas } from 'react-three-fiber'

A friend was nice enough to design a couple mockups for the
new [MDX](https://mdxjs.com) splash page and I fell in love
with his floating shape concepts.

![mdxjs.com mockup with shapes](https://c8r.imgix.net/4804cd1131c8823914e19880/image.png)

Since MDX is a content format I wanted the "visual treatment"
to be minimal and I felt like these shapes were a bit retro
which fit in nicely with the slightly-brutalist aesthetic
of the website.

Below is my process of building out the shapes using
[react-three-fiber](https://github.com/drcmda/react-three-fiber).

export const Step1 = ({ vertices = [[-1, 0, 0], [0, 1, 0], [1, 0, 0], [0, -1, 0], [-1, 0, 0]] }) => (
  <group>
    <line>
      <geometry
        attach="geometry"
        vertices={vertices.map(v => new THREE.Vector3(...v))}
      />
    </line>
    <mesh>
      <octahedronGeometry attach="geometry" />
      <meshBasicMaterial attach="material" color="tomato" transparent />
    </mesh>
  </group>
)
